strand.foldLeft(("", output)) { (tup, chr) =>
  if (tup._2.contains(chr)) {
    (tup._1, tup._2 + (chr -> (tup._2(chr) + 1)))
  } else (s"invalid nucleotide '$chr'", tup._2)
} match {
  case (errStr, output) if errStr.isEmpty() => Right(output)
  case (errStr, _)                          => Left(errStr)
}
